    strFlag = False
    while strFlag == False:
        strEmployeeStatus = str(input("Enter employee status (management or hourly): "))
        strFlag, strEmployeeStatus = ValidateEmployeeStatus(strEmployeeStatus)
        if strFlag == True:
            intYearsEmployed = input("Enter years employed: ")
            strFlag, intYearsEmployed = ValidateIntInput(intYearsEmployed)
            if strFlag == True:
                decYTDPurchases = input("Enter total amount of previous purchases before discount: ")
                strFlag, decYTDPurchases = ValidateDecInput(decYTDPurchases)
                if strFlag == True:
                    decSubtotal = input("Enter today's purchase total: ")
                    strFlag, decSubtotal = ValidateDecInput(decSubtotal)
                    if strFlag == True:
                        return strEmployeeStatus, intYearsEmployed, decYTDPurchases, decSubtotal

  2. Order Discount Capped at Annual Limit

    This pricing function applies a computed order discount but enforces a $200 yearly cap. It is a compact example of business-rule enforcement with conditional guards. 

    def CalculateOrderDiscount(decSubtotal, decDiscountPercent, decYTDDiscount):
    decOrderDiscount = decSubtotal * decDiscountPercent

    if decYTDDiscount > 200.00:
        decOrderDiscount = 0.00
    else:
        if decYTDDiscount + decOrderDiscount > 200.00:
            decOrderDiscount = 200.00 - decYTDDiscount

    return decOrderDiscount
